## Usage

### Search Variant
<Preview>
  <Story 
    name='Search Variant'
    play={({ canvasElement }) => {
        var dataSource = [
            { value: '1', text: 'Luffy' },
            { value: '2', text: 'Zorro' },
            { value: '3', text: 'Sanji' },
        ];
        var standardVariant = document.getElementById('standardVariant');
        standardVariant.options = dataSource;
        standardVariant.value = ['2'];
        standardVariant.multiple = true;
    }}
  >
  {() => `<fw-list-options
    id="standardVariant"
    slot="popover-content"
    searchable="true"
    not-found-text="Pirate not found"
    search-text="Search Pirate"
  ></fw-list-options>`}
  </Story>
</Preview>
