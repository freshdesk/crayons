## Usage

### Default
<Preview>
  <Story 
    name='Default'
    play={({ canvasElement }) => {
      const timer = setTimeout(()=>{
        document.querySelector('fw-tooltip').show();
        clearTimeout(timer);
      }, 500);
    }}
  >
    {
      () => `
        <fw-tooltip placement="right" content="This is the first tooltip example">
          <fw-button> Show tooltip </fw-button>
        </fw-tooltip>
      `
    }
  </Story>
</Preview>
