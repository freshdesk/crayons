## datepicker open
<Preview>
  <Story 
    name='Default'
    play={ async ({ canvasElement }) => {
      setTimeout(()=>{
        const datePicker = document.querySelector('fw-datepicker');
        datePicker.focus();
        datePicker.shadowRoot.querySelector('fw-popover').querySelector('.datepicker').style.display="block";
        datePicker.shadowRoot.querySelector('fw-popover').querySelector('.daterangepicker').style.display="none";
        datePicker.shadowRoot.querySelector('fw-popover').querySelector('fw-input').click();    
      },500);
    }}
    decorators={[
      (Story) => {
        const story = Story();
        return `<div>${story}</div>`;
      }
    ]}
  >
    {Template.bind({})}
  </Story>
</Preview>
